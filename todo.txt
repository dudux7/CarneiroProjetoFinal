processar_agendamento.php


<?php
require_once 'conexao.php';

if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    // Coleta os dados do formulário
    $idServico = $_POST['idServico'];
    $idData = $_POST['id_data'];
    $idHorario = $_POST['id_horario'];
    $observacoes = $_POST['observacoes'];

    // Verifica se todos os campos estão preenchidos
    if (!empty($idServico) && !empty($idData) && !empty($idHorario)) {
        // Inserir agendamento
        $query = "INSERT INTO agendamentos (datahora, idCliente, idServico, id_data, id_horario, observacoes, status) 
                  VALUES (NOW(), 1, ?, ?, ?, ?, 'pendente')";
        $stmt = $conexao->prepare($query);
        $stmt->bind_param("iiis", $idServico, $idData, $idHorario, $observacoes);

        if ($stmt->execute()) {
            echo "Agendamento realizado com sucesso!";
        } else {
            echo "Erro ao agendar: " . $stmt->error;
        }

        $stmt->close();
    } else {
        echo "Por favor, preencha todos os campos!";
    }

    mysqli_close($conexao);
}
?>
